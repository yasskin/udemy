    <form> First Name :
        <input type="text" name="e1" value="Laurence" onchange="cFun()">
        <br> Last Name :
        <input type="text" name="e2" value="Svekis" onblur="cFun()">
        <br>
        <input type="button" onClick="sendInfo()" value="Submit"> </form>
    <script>
        function cFun() {
            var p = event.target;
            if (p.name == "e1") {
                message("First Name Changed to  " + p.value);
            }
            else {
                message("Last Name Changed to  " + p.value);
            }
        }

        function sendInfo() {
            var p = event.target.parentElement;
            message("Welcome " + p.e1.value + " " + p.e2.value);
        }

        function message(m) {
            document.getElementById('wrapper').innerHTML = m;
        }
    </script>
