        function cCookie(name, value, days) {
            var e = "";
            if (days) {
                var d = new Date();
                d.setTime(d.getTime() + (days * 24 * 60 * 60 * 1000));
                e = "; expires=" + d.toUTCString();
            }
            document.cookie = name + "=" + value + e + "; path=/";
        }
